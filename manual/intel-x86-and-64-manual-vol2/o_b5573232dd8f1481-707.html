Operation

MIN(SRC1, SRC2)
{

IF ((SRC1 = 0.0) and (SRC2 = 0.0)) THEN DEST SRC2;

ELSE IF (SRC1 = SNaN) THEN DEST SRC2; FI;
ELSE IF (SRC2 = SNaN) THEN DEST SRC2; FI;
ELSE IF (SRC1 < SRC2) THEN DEST SRC1;
ELSE DEST SRC2; 

FI; 

}
